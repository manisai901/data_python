    'email': 'fake.email',
    'business_email': 'fake.company_email',
    'website': 'fake.url'

# Generate BigQuery schema
schema = [
    {
        "name": column,
        "type": "STRING",
        "mode": "NULLABLE"
    }
    for column in GENERATORS

# Save to a JSON file

    json.dump(schema, f, indent=2)
